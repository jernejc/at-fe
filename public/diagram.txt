    %% Entry Point
    subgraph entry_group[" API Gateway "]
        gateway_node["Processing Gateway"]
    end

    %% Orchestration Layer
    subgraph orchestrator_group[" Orchestration Layer "]
        agentic_company_orchestrator["Company Orchestrator"]
    end

    %% Processing Agents - ordered for optimal alignment
    %% Data Collector in center as it connects to all data sources
    subgraph service_group[" Processing Agents "]
        direction TB
        agentic_fit_calculator["Fit Calculator"]
        agentic_signal_analyzer["Signal Analyzer"]
        agentic_data_collector["Data Collector"]
        agentic_playbook_generator["Playbook Generator"]
        market_analyst["Market Analyst"]
    end

    %% Data Sources - ordered to align with service agent connections
    %% database centered as all service agents connect to it
    subgraph data_group[" Data Sources "]
        direction TB
        coresignal["Coresignal"]
        database["Database"]
        discovery["Discovery"]
        llm_extraction["LLM Extraction"]
    end

    %% Request Flow - Main pipeline
    gateway_node ==> agentic_company_orchestrator

    %% Orchestrator to Service Agents
    agentic_company_orchestrator --> agentic_data_collector
    agentic_company_orchestrator --> agentic_fit_calculator
    agentic_company_orchestrator --> agentic_playbook_generator
    agentic_company_orchestrator --> agentic_signal_analyzer
    agentic_company_orchestrator --> market_analyst

    %% Service Agents to Data Sources

    agentic_data_collector --> database
    agentic_data_collector --> discovery
    agentic_data_collector --> llm_extraction
    agentic_fit_calculator --> database
    agentic_playbook_generator --> database
    agentic_signal_analyzer --> database
    market_analyst --> database
